<script>
    window.addEventListener('load', function() {
        const navLinks = document.querySelectorAll('.nav-link');
        const sections = document.querySelectorAll('main section');

        const observer = new IntersectionObserver((entries) => {
            entries.forEach(entry => {
                if (entry.isIntersecting) {
                    navLinks.forEach(link => {
                        link.classList.remove('active-link');
                        if (link.getAttribute('href').substring(1) === entry.target.id) {
                            link.classList.add('active-link');
                        }
                    });
                }
            });
        }, { threshold: 0.5 });

        sections.forEach(section => observer.observe(section));
        
        navLinks.forEach(link => {
            link.addEventListener('click', (e) => {
                e.preventDefault();
                document.querySelector(link.getAttribute('href')).scrollIntoView({
                    behavior: 'smooth'
                });
            });
        });

        const accordions = document.querySelectorAll('.accordion-item');
        accordions.forEach(item => {
            const button = item.querySelector('.accordion-button');
            const content = item.querySelector('.accordion-content');
            const icon = button.querySelector('span:last-child');

            button.addEventListener('click', () => {
                const isExpanded = content.style.maxHeight && content.style.maxHeight !== '0px';
                if (isExpanded) {
                    content.style.maxHeight = '0px';
                    content.style.paddingTop = '0';
                    content.style.paddingBottom = '0';
                    icon.style.transform = 'rotate(0deg)';
                } else {
                    content.style.maxHeight = content.scrollHeight + 'px';
                    content.style.paddingTop = '0.5rem';
                    content.style.paddingBottom = '0.5rem';
                    icon.style.transform = 'rotate(180deg)';
                }
            });
        });
        
        const citationButtons = document.querySelectorAll('.citation-btn');
        const citationContent = document.getElementById('citation-content');

        const citationData = {
            'indireta': {
                title: 'Citação Indireta (A mais comum!)',
                description: 'Você lê a ideia do autor e a reescreve com as <strong>suas palavras</strong>.',
                example: `<p class="font-medium">Formato:</p><p class="font-mono text-sm">SOBRENOME (ANO) ou (SOBRENOME, ANO)</p><div class="citation-example"><p><strong>Exemplo 1:</strong> Conforme Silva (2023), a tecnologia é um pilar essencial para o avanço da sociedade.</p><p class="mt-2"><strong>Exemplo 2:</strong> A tecnologia é um pilar essencial para o avanço da sociedade (SILVA, 2023).</p></div>`
            },
            'direta-curta': {
                title: 'Citação Direta Curta (até 3 linhas)',
                description: 'Você copia exatamente um trecho e o insere no seu parágrafo, entre <strong>"aspas duplas"</strong>.',
                 example: `<p class="font-medium">Formato:</p><p class="font-mono text-sm">(SOBRENOME, ANO, p. X)</p><div class="citation-example"><p><strong>Exemplo:</strong> A afirmação de que "...a tecnologia é fundamental para a sociedade" (SILVA, 2023, p. 15) reforça a necessidade de seu uso ético.</p></div>`
            },
            'direta-longa': {
                title: 'Citação Direta Longa (mais de 3 linhas)',
                description: 'O trecho é apresentado em um parágrafo separado, com formatação especial.',
                 example: `<ul class="list-disc list-inside text-sm space-y-1 my-2"><li>Recuo de <strong>4 cm</strong> da margem esquerda.</li><li>Fonte <strong>menor</strong> (tamanho 10).</li><li><strong>Sem aspas</strong>.</li><li>Espaçamento entre linhas <strong>simples</strong>.</li></ul><div class="citation-example"><p class="font-medium mb-2">Exemplo 1 (Autor no final):</p><div class="long-citation-example">Todas essas fortes tendências influenciam poderosamente as organizações e seu estilo de administrar as pessoas. O mundo dos negócios ficou completamente diferente, exigente, dinâmico, mutável e incerto. E as pessoas sentem o impacto dessas influências e necessitam de um apoio e suporte por parte dos seus líderes e gerentes. (CHIAVENATO, 2004, p. 42)</div></div><div class="citation-example"><p class="font-medium mb-2">Exemplo 2 (Autor no início):</p><p>Para Chiavenato (2004, p. 42):</p><div class="long-citation-example mt-2">Todas essas fortes tendências influenciam poderosamente as organizações e seu estilo de administrar as pessoas. E as pessoas sentem o impacto dessas influências e necessitam de um apoio e suporte por parte dos seus líderes e gerentes.</div></div>`
            }
        };
        
        function updateCitationContent(type) {
            const data = citationData[type];
            citationContent.innerHTML = `<h4 class="font-bold text-xl text-gray-900">${data.title}</h4><p class="mt-2 text-gray-600">${data.description}</p><div class="mt-4">${data.example}</div>`;
            citationButtons.forEach(btn => {
                if (btn.dataset.type === type) {
                    btn.classList.remove('bg-gray-200', 'text-gray-700', 'hover:bg-gray-300');
                    btn.classList.add('btn-primary');
                } else {
                    btn.classList.add('bg-gray-200', 'text-gray-700', 'hover:bg-gray-300');
                    btn.classList.remove('btn-primary');
                }
            });
        }

        citationButtons.forEach(button => button.addEventListener('click', () => updateCitationContent(button.dataset.type)));
        updateCitationContent('indireta');

    });

    function checkQuiz(type) {
        if (type === 'indireta') {
            const answerEl = document.getElementById('answer-indireta');
            answerEl.innerHTML = '<strong>Exemplo:</strong> Segundo Ferreira (2024), a educação é capaz de modificar tanto o ser humano quanto a sociedade.';
            answerEl.classList.remove('hidden');
        } else if (type === 'direta') {
            const answerEl = document.getElementById('answer-direta');
            answerEl.innerHTML = '<strong>Exemplo:</strong> De acordo com o autor, "a educação transforma o indivíduo e a coletividade" (FERREIRA, 2024, p. 42).';
            answerEl.classList.remove('hidden');
        }
    }
